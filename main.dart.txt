import 'package:flutter/material.dart';

void main() => runApp(ElderlyCareApp());

class ElderlyCareApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) => MaterialApp(
        title: 'CompanionCare',
        theme: ThemeData(
          colorScheme: ColorScheme.fromSwatch(primarySwatch: Colors.teal),
          useMaterial3: true,
        ),
        home: WelcomeScreen(),
        routes: appRoutes,
      );

  static final appRoutes = <String, WidgetBuilder>{
    '/home': (context) => HomeScreen(),
    '/checkin': (context) => CheckInScreen(),
    '/music_playlist': (context) => MusicPlaylistScreen(),
    '/memory_lane': (context) => MemoryLaneScreen(),
    '/contacts/emergency': (context) =>
        ContactsScreen(featureTitle: 'Emergency', nextRoute: '/emergency'),
    '/contacts/video': (context) =>
        ContactsScreen(featureTitle: 'Video Call', nextRoute: '/video'),
    '/contacts/support': (context) =>
        ContactsScreen(featureTitle: 'Support', nextRoute: '/support'),

    '/medication': (context) => MedicationScreen(),
    '/emergency': (context) => EmergencyScreen(),
    '/activity': (context) => ActivityLogScreen(),
    '/settings': (context) => SettingsScreen(),
    '/support': (context) => SupportScreen(),
    '/welcome': (context) => WelcomeScreen(),
  };
}

class HomeScreen extends StatelessWidget {
  final Map<String, String?> contactManagementRoutes = {
    'Emergency': '/contacts/emergency',
    'Video Call': '/contacts/video',
    'Support': '/contacts/support',
  };

  final List<Map<String, dynamic>> features = [
    {'title': 'Check-in', 'route': '/checkin', 'icon': Icons.check_circle_outline},
    {'title': 'Video Call', 'route': '/video', 'icon': Icons.video_call},
    {'title': 'Medication', 'route': '/medication', 'icon': Icons.medical_services},
    {'title': 'Emergency', 'route': '/emergency', 'icon': Icons.local_hospital},
    {'title': 'Activity', 'route': '/activity', 'icon': Icons.directions_run},
    {'title': 'Settings', 'route': '/settings', 'icon': Icons.settings},
    {'title': 'Support', 'route': '/support', 'icon': Icons.support_agent},
  ];

  @override
  Widget build(BuildContext context) => Scaffold(
        appBar: AppBar(
          title: Text('Welcome to CompanionCare',
              style: TextStyle(color: Colors.white)),
          backgroundColor: Colors.teal,
          iconTheme: IconThemeData(color: Colors.white),
        ),
        backgroundColor: Colors.teal[50],
        body: GestureDetector(
          onHorizontalDragEnd: (details) {
            if (details.primaryVelocity! > 0) {
              Navigator.pushNamed(context, '/music_playlist');
            } else if (details.primaryVelocity! < 0) {
              Navigator.pushNamed(context, '/memory_lane');
            }
          },
          child: Stack(
            children: [
              Padding(
                padding: const EdgeInsets.all(6.0),
                child: GridView.builder(
                  gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
                    crossAxisCount: 3,
                    mainAxisSpacing: 6,
                    crossAxisSpacing: 6,
                    childAspectRatio: 1,
                  ),
                  itemCount: features.length,
                  itemBuilder: (context, index) {
                    final feature = features[index];
                    final contactRoute =
                        contactManagementRoutes[feature['title']];

                    return ElevatedButton(
                      style: ElevatedButton.styleFrom(
                        padding: EdgeInsets.all(8),
                        backgroundColor: feature['title'] == 'Emergency'
                            ? Colors.red
                            : Colors.teal,
                        shape: RoundedRectangleBorder(
                            borderRadius: BorderRadius.circular(10)),
                      ),
                      onPressed: () {
                        if (contactRoute != null) {
                          Navigator.pushNamed(context, contactRoute);
                        } else {
                          Navigator.pushNamed(
                              context, feature['route'] as String);
                        }
                      },
                      child: Column(
                        mainAxisAlignment: MainAxisAlignment.center,
                        children: [
                          Icon(feature['icon'] as IconData?,
                              size: 30, color: Colors.white),
                          SizedBox(height: 4),
                          Text(
                            feature['title'] as String,
                            style: TextStyle(
                                fontSize: 16,
                                fontWeight: FontWeight.bold,
                                color: Colors.white),
                            textAlign: TextAlign.center,
                          ),
                        ],
                      ),
                    );
                  },
                ),
              ),
              Positioned(
                bottom: 20,
                right: 20,
                child: FloatingActionButton(
                  heroTag: "music_fab",
                  backgroundColor: Colors.teal,
                  onPressed: () =>
                      Navigator.pushNamed(context, '/music_playlist'),
                  child: Icon(Icons.music_note, color: Colors.white),
                ),
              ),
              Positioned(
                bottom: 20,
                left: 20,
                child: FloatingActionButton(
                  heroTag: "memory_fab",
                  backgroundColor: Colors.teal,
                  onPressed: () => Navigator.pushNamed(context, '/memory_lane'),
                  child: Icon(Icons.image, color: Colors.white),
                ),
              ),
            ],
          ),
        ),
      );
}

class ContactsScreen extends StatefulWidget {
  final String featureTitle;
  final String nextRoute;

  ContactsScreen({required this.featureTitle, required this.nextRoute});

  @override
  _ContactsScreenState createState() => _ContactsScreenState();
}

class _ContactsScreenState extends State<ContactsScreen> {
  List<Contact> contacts = [];
  final _nameController = TextEditingController();
  final _phoneController = TextEditingController();

  void _addContact() {
    final name = _nameController.text.trim();
    final phone = _phoneController.text.trim();
    if (name.isEmpty || phone.isEmpty) {
      showSnackBar('Please enter both name and phone number');
      return;
    }
    setState(() {
      contacts.add(Contact(name: name, phone: phone));
      _nameController.clear();
      _phoneController.clear();
    });
    FocusScope.of(context).unfocus();
  }

  void _removeContact(int index) => setState(() => contacts.removeAt(index));

  void showSnackBar(String message) => ScaffoldMessenger.of(context)
      .showSnackBar(SnackBar(content: Text(message)));

  @override
  void dispose() {
    _nameController.dispose();
    _phoneController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) => Screen(
        screenTitle: '${widget.featureTitle} Contacts',
        body: Padding(
          padding: EdgeInsets.all(16),
          child: Column(
            children: [
              Text('Add contacts for ${widget.featureTitle.toLowerCase()}',
                  style: subtitleTextStyle()),
              SizedBox(height: 8),
              TextField(
                controller: _nameController,
                decoration: inputDecoration('Name'),
                textInputAction: TextInputAction.next,
              ),
              SizedBox(height: 8),
              TextField(
                controller: _phoneController,
                decoration: inputDecoration('Phone Number'),
                keyboardType: TextInputType.phone,
                textInputAction: TextInputAction.done,
                onSubmitted: (_) => _addContact(),
              ),
              SizedBox(height: 8),
              StyledButton(
                icon: Icons.add,
                text: 'Add Contact',
                onPressed: _addContact,
              ),
              SizedBox(height: 20),
              Expanded(
                child: contacts.isEmpty
                    ? Center(
                        child: Text('No contacts added yet.',
                            style: itemTextStyle()),
                      )
                    : ListView.builder(
                        itemCount: contacts.length,
                        itemBuilder: (context, index) => ContactCard(
                            contact: contacts[index],
                            onDelete: () => _removeContact(index)),
                      ),
              ),
              StyledButton(
                text: 'Proceed to ${widget.featureTitle}',
                onPressed: () {
                  if (contacts.isEmpty) {
                    showSnackBar('Add at least one contact to proceed');
                    return;
                  }
                  Navigator.pushNamed(context, widget.nextRoute);
                },
              ),
            ],
          ),
        ),
      );

  InputDecoration inputDecoration(String label) => InputDecoration(
        labelText: label,
        border: OutlineInputBorder(),
      );
}

class Contact {
  final String name;
  final String phone;
  Contact({required this.name, required this.phone});
}

class Screen extends StatelessWidget {
  final String? screenTitle;
  final Widget? body;
  final Color? backgroundColor;
  final Widget? child;

  const Screen({
    Key? key,
    this.screenTitle,
    this.body,
    this.backgroundColor,
    this.child,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) => Scaffold(
        appBar: screenTitle != null
            ? AppBar(
                title: Text(screenTitle!, style: TextStyle(color: Colors.white)),
                backgroundColor: Colors.teal,
                iconTheme: IconThemeData(color: Colors.white),
              )
            : null,
        backgroundColor: backgroundColor ?? Colors.teal[50],
        body: body ?? child,
      );
}

class StyledButton extends StatelessWidget {
  final IconData? icon;
  final String text;
  final VoidCallback onPressed;
  final Color? backgroundColor;

  const StyledButton({
    Key? key,
    this.icon,
    required this.text,
    required this.onPressed,
    this.backgroundColor = Colors.teal,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) => ElevatedButton.icon(
        icon: Icon(icon, color: Colors.white),
        label: Text(text, style: TextStyle(fontSize: 18, color: Colors.white)),
        style: ElevatedButton.styleFrom(
          backgroundColor: backgroundColor,
          padding: EdgeInsets.symmetric(horizontal: 40, vertical: 15),
          shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
          minimumSize: Size(double.infinity, 50),
        ),
        onPressed: onPressed,
      );
}

    fontSize: 18, fontWeight: FontWeight.w600, color: Colors.teal[900]);
